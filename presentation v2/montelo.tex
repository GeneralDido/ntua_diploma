\section{Μοντελοποίηση}

\subsection*{?}
\begin{frame}
\begin{center}
\textbf{\tl{SMP}} \\
Τιμή ενέργειας στην χονδρεμπορική αγορά
\end{center}
\end{frame}

\subsection*{?}
\begin{frame}
\frametitle{Παράγοντες που επηρεάζουν το \tl{SMP}}
\begin{enumerate}
\item Η ζήτηση της ηλεκτρικής ενέργειας
\item Η παραγωγή από ΑΠΕ
\item Οι τιμές των καυσίμων
\item Η διαθεσιμότητα των πόρων
\end{enumerate}
\end{frame}

\subsection*{?}
\begin{frame}
\frametitle{Σκοπός}
\begin{itemize}
\item Δημιουργία μοντέλου για την πρόβλεψη του \tl{SMP} με βάση τις παραμέτρους που το επηρεάζουν
\item Κατανόηση των σχέσεων μεταξύ των μεταβλητών του μοντέλου (\tl{predictors},$\mathbf{X}$) και του \tl{SMP} (έξοδος - \tl{response},$Y$)
\end{itemize}
\end{frame}

\subsection*{?}
\begin{frame}
\frametitle{Παραδοχές για το μοντέλο}
\begin{enumerate}
  \item Δεν υπάρχει αιτιώδης συνάφεια μεταξύ της εξόδου (\tl{SMP}) και ενός ή περισσοτέρων μεταβλητών εισόδου. 
% Οι μεταβλητές είναι εξωγενείς υπό την έννοια ότι δεν υπάρχει αντίστροφη αιτιώδης συνάφεια από την εξαρτημένη μεταβλητή σε έναν ή περισσότερους predictors
  \item Χρησιμοποιούνται όλες οι σχετικές με το μοντέλο μεταβλητές
  \item Οι περιττές επεξηγηματικές μεταβλητές που προσθέτουν θόρυβο στο σύστημα αποκλείονται. 
  \item Αποφεύγεται η συμπερίληψη στο μοντέλο πολλαπλών συσχετισμένων μεταξύ τους μεταβλητών.
\end{enumerate}
\end{frame}

\subsection*{?}
\begin{frame}
\frametitle{Μεταβλητές Μοντέλου}
\begin{description}
  \item[\textbf{\tl{smp}}] Τιμή ενέργειας στην χονδρεμπορική αγορά
  \item[\tl{availability}] Δυναμικότητα μονάδων παραγωγής λιγνίτη 
  \item[\tl{exports}] Εξαγωγές ηλεκτρικής ενέργειας 
  \item[\tl{hydrogen}] Παραγωγή υδροηλεκτρικών εργοστασίων 
  \item[\tl{imports}]  Εισαγωγές ηλεκτρικής ενέργειας 
  \item[\tl{lignite}]  Παραγωγή λιγνίτη
  \item[\tl{load\_forecast}] Ζήτηση ηλεκτρικής ενέργειας
  \item[\tl{ngas}] Παραγωγή ενέργειας από φυσικό αέριο
  \item[\tl{res\_forecast}] Πρόγνωση παραγωγής ΑΠΕ
  \item[\tl{waters}] Παραγωγή υποχρεωτικών νερών
  \item[\tl{waip}] Τιμή φυσικού αερίου
\end{description}
Μονάδα ενέργειας: \tl{MW} \\ Δεδομένα: Ημερήσια 1/1/2009 - 30/11/2013
\end{frame}
